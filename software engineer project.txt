#include<iostream>

struct student {

	string location;
	string month;
	student* next;

	student(string i = "", string s = "", student* n = nullptr)
	{
		location = i;
		month = s;
		next = n;

	}

};

void printlist(student*);
void add_node_end(student*&, string, string);

void printlist(student* head)
{

	student* temp = head;

	if (!head)
		cout << "Empty linked list, Nothing to print\n";
	else
		while (temp)
		{
			cout << temp->location << " " << temp->month ;
			cout << "\n";
			temp = temp->next;

		}

}

void add_node_end(student*& head, string i, string s)
{

	if (!head) // empty linked list
		head = new student(i, s);
	else
	{
		student* temp = head;
		while (temp->next != nullptr)
			temp = temp->next;
		temp->next = new student(i, s);

	}
}
